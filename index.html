Pour connecter votre Google Sheet, suivez ces étapes :

  1. Ouvrez votre Google Sheet
  2. Cliquez sur "Extensions" > "Apps Script"
  3. Copiez-collez ce code dans l'éditeur :
function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Joueurs"); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ error: "Feuille 'Joueurs' introuvable" })) .setMimeType(ContentService.MimeType.JSON); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const players = []; for (let i = 1; i < data.length; i++) { const player = {}; for (let j = 0; j < headers.length; j++) { player[headers[j]] = data[i][j]; } players.push(player); } return ContentService.createTextOutput(JSON.stringify(players)) .setMimeType(ContentService.MimeType.JSON); }
